أوامر الإعداد الأولي
git config --global user.name "اسمك" - لتعيين اسم المستخدم

git config --global user.email "بريدك@example.com" - لتعيين البريد الإلكتروني

git config --list - لعرض إعدادات Git

أوامر إنشاء وإدارة المستودعات
git init - لبدء مستودع جديد في المجلد الحالي

git clone <رابط-المستودع> - لاستنساخ مستودع موجود

أوامر تتبع التغييرات
git status - لعرض حالة الملفات

git add <اسم-الملف> - لإضافة ملف للتتبع

git add . - لإضافة جميع الملفات

git commit -m "رسالة توضح التغييرات" - لحفظ التغييرات مع رسالة

git commit -am "رسالة التوضيح" - لإضافة وcommit في أمر واحد

أوامر الفروع (Branches)
git branch - لعرض الفروع

git branch <اسم-الفرع> - لإنشاء فرع جديد

git checkout <اسم-الفرع> - للانتقال إلى فرع

git checkout -b <اسم-الفرع> - لإنشاء فرع والانتقال إليه مباشرة

git merge <اسم-الفرع> - لدمج فرع مع الفرع الحالي

git branch -d <اسم-الفرع> - لحذف فرع

أوامر العمل مع Remote
git remote add origin <رابط-المستودع> - لربط المستودع المحلي بالبعيد

git push origin <اسم-الفرع> - لرفع التغييرات

git pull origin <اسم-الفرع> - لجلب التغييرات من البعيد

git remote -v - لعرض الـ remotes

أوامر عرض السجل والتغييرات
git log - لعرض سجل الـ commits

git diff - لعرض التغييرات في الملفات

git diff <اسم-الملف> - لعرض التغييرات لملف معين

أوامر التراجع عن التغييرات
git checkout -- <اسم-الملف> - للتراجع عن التعديلات في ملف (قبل الـ add)

git reset HEAD <اسم-الملف> - لإلغاء إضافة ملف (بعد الـ add وقبل الـ commit)

git revert <كود-الcommit> - للتراجع عن commit محدد
